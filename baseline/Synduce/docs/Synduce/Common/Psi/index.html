Module Common.Psi

type t = {
id : Base.int;(*

An identifier for the problem.

*)
target : Lang.PMRS.t;(*

The target recursion skeleton in the problem.

*)
reference : Lang.PMRS.t;(*

The reference function in the problem.

*)
repr : Lang.PMRS.t;(*

The representation function in the problem.

*)
tinv : Lang.PMRS.t Base.option;(*

The requires predicate of the target recursion skeleton.

*)
repr_is_identity : Base.bool;(*

A boolean that is true iff psi_repr is identity.

*)
lifting : Lang.RType.t Base.list;(*

The current lifting: the output type of psi_target should be !_alpha extended with the tuple of types psi_lifting.

*)
}

PsiDef.t is the definition of the problem: the goal is to synthesize the unknowns in target such that target = orig ∘ repr

val logics : t -> Lang.SmtLogic.logic_info list
val compare : t -> t -> int
val equal : t -> t -> bool
val hash : t -> Base.int
val psi_id : Base.int Base.ref
val new_psi_id : unit -> Base.int
