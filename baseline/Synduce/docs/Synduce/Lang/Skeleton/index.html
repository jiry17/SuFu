| SChoice of t Base.list(*

A choice of possible guesses.

*)
| SBin of Term.Binop.t * t * t(*

A binary expression.

*)
| SUn of Term.Unop.t * t(*

A unary expression.

*)
| SIte of t * t * t(*

An if-then-else.

*)
| SType of RType.t(*

A guess of some type (to be filled with the appropriate non-terminal).

*)
| STypedWith of RType.t * t Base.list
| SArg of Base.int(*

A direct reference to a function argument.

*)
| STuple of t Base.list
| SNonGuessable
